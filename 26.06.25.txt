// 1~10 더하기,  짝수함, 홀수함.
// for 문과 변수정의의 위치에 주의.

	var total = 0;
	var jjack = 0;
	var hol = 0;
	for (var k=1; k<=10; k++){
		total +=k;
		if(k%2==0){
			jjack +=k;
		}
		else{
			hol +=k;
		}
	}
		document.write("total : "+total+"<br>");
		document.write("total : "+jjack+"<br>");
		document.write("total : "+hol+"<br>");
</script>
