	<script>
	var name = prompt("성명?");
	var kor = prompt("국어점수는?");  
	var eng = prompt("영어 점수는?"); 
	var mat = prompt("수학 점수는?");  
	var total = Number(kor)+Number(eng)+Number(mat);
	var ave = (total/3) - ((total/3)%1);
	document.write(name+" 님의 총점은 "+total+"점<br>");
	document.write(name+" 님의 평균은 "+ave+"점");
	</script>

	var total = prompt("점수는?");
	if (Number(total)>=80) {
		document.write("1.합격");

		document.write("2.불합격");

	<script>									//else 는 if 문과 짝으로 참이면 
		var point = prompt("점수는?");			//if 문 안의 조건은 참,거짓이 판명가능한 조건만 가능하다. 
		if (Number(point)>=90) {					//참이면 {} 실행, 거짓이면 작동하지않는다.
			document.write("a");				// if(ave=90) 불가능,  if(ave==90) 가능.
		}
		else {
			if (Number(point)>=80) {
				document.write("b");
			}
			else{
				if (Number(point)>=70) {
					document.write("c");
				}
				else{
					document.write("F");
				}
			}
			}
		
		}
