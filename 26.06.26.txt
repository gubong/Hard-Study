// 과목 수 만큼 점수 입력. > 입력한 것 출력, 와일문 다 끝나면 총점출력.
<script>
	var sub = Number(prompt(몇 과목?));
	var total = 0;
	
	var k =1;
	while(k<=sub){
		var point = Number(prompt(몇 점?));
		document.write(point+"<br>");
		total += point;
		k++	
	} 
	
	var ave = total/sub;
	document.write(total+"<br>");
	document.write(ave);
	
</script>

// 홍길동, 차길동, 박길동, 이명박, 표길동
var count = Number(prompt("몇 명?"));
var info=[];

for (k=0;k<count; k++){
	var info[k] = prompt(k+1+"번째 이름은?");		
}
var search = prompt("어떤 이름 검색할래?");    //박 검색
for (k=0; k<info.length; k++){
	if (info[k].indexOf(search) != -1){
		document.write(k+1+"번째 :"+info[k]+"<br>");	
	}
}
